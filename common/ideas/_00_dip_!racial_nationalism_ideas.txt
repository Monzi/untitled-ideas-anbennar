	category = DIP
	trigger = {
		NOT = {
			has_idea_group = assimilation_ideas
		}
	}

	bonus = {
		culture_conversion_cost = -0.25
		culture_conversion_time = -0.15
		monthly_cultural_unity = 0.5
	}

	nationalism_ideas_1 = {
		hostile_attrition = 1
	}
	nationalism_ideas_2 = {
		manpower_in_culture_group_provinces = 0.10
		same_culture_advisor_cost = -0.15
	}
	nationalism_ideas_3 = {
		reinforce_cost_modifier = -0.2
	}
	nationalism_ideas_4 = {
		years_of_nationalism = -5
	}
	nationalism_ideas_5 = {
		land_morale = 0.1
	}
	nationalism_ideas_6 = {
		global_autonomy = -0.05
	}
	nationalism_ideas_7 = {
		# cb_early_nationalist = yes
		# Gives an early and nerfed Nationalism CB.
		province_warscore_cost = -0.1
	}

	ai_will_do = {
		factor = 0.50

		modifier = {
			factor = 3
			personality = ai_militarist
		}
		modifier = {
			factor = 2
			culture_group_claim = from
		}
	}
}
